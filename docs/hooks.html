Legacy Source Injection

For pre-GKI kernels (4.14, 4.19, 5.4), NexusSU relies on direct source tree injection. By modifying the kernel's core execution and file system files before compilation, we achieve an embedded, undetectable root environment.

Step 1: The Core Header (include/linux/nexussu.h)

This header acts as the central brain. It defines the privilege escalation function that swaps the current process credentials to UID 0 (Root). This file is placed in the include/linux/ directory of the kernel tree.

+ include/linux/nexussu.h 

#ifndef _NEXUSSU_H
#define _NEXUSSU_H

#include <linux/cred.h>
#include <linux/uidgid.h>

/* Function prototypes exposed by the state engine */
extern bool nexussu_is_granted(kuid_t uid);
extern bool nexussu_stealth_check(const char __user *filename);
extern void nexussu_scrub_proc_buffer(struct file *file, char __user *buf, size_t count, ssize_t *ret);

/* 
 * Core escalation function.
 * Prepares a new credential block and commits global root IDs 
 * to elevate the calling process.
 */
static inline void nexussu_escalate(void) {
    struct cred *new_creds;
    
    new_creds = prepare_creds();
    if (new_creds != NULL) {
        /* Assign Global Root IDs */
        new_creds->uid = GLOBAL_ROOT_UID;
        new_creds->gid = GLOBAL_ROOT_GID;
        new_creds->euid = GLOBAL_ROOT_UID;
        new_creds->egid = GLOBAL_ROOT_GID;
        new_creds->suid = GLOBAL_ROOT_UID;
        new_creds->sgid = GLOBAL_ROOT_GID;
        new_creds->fsuid = GLOBAL_ROOT_UID;
        new_creds->fsgid = GLOBAL_ROOT_GID;
        
        /* Grant full capability sets to bypass primitive environment boundaries */
        new_creds->cap_permitted = CAP_FULL_SET;
        new_creds->cap_effective = CAP_FULL_SET;
        new_creds->cap_bset = CAP_FULL_SET;
        new_creds->cap_ambient = CAP_FULL_SET;
        
        /* Commit the elevated credentials back to the task */
        commit_creds(new_creds);
    }
}

#endif /* _NEXUSSU_H */

Step 3: VFS Directory Stealth (fs/readdir.c)

Granting root is only half the battle; hiding it is the rest. We intercept the Virtual File System's directory-reading capabilities to hide our root binaries.

 */

#include <linux/string.h>

#define HIDE_TARGET "su"

 * Our custom wrapper for the directory filler.
 * It checks the name of the file being read. If it matches our target,
 * it returns 0 (success) without actually writing the entry to the user's buffer.
 */
static int nexussu_filldir_hook(struct dir_context *ctx, const char *name, int namlen,
                                loff_t offset, u64 ino, unsigned int d_type)
{
    /* If the file is 'su', skip it completely */
    if (namlen == 2 && strncmp(name, HIDE_TARGET, 2) == 0) {
        printk(KERN_DEBUG "NexusSU: Hid '%s' from directory listing\n", HIDE_TARGET);
        return 0; 
    }
    return 0;
}

Step 4: Control Interface (drivers/char/nexussu.c)

A standalone kernel root engine requires a communication bridge to receive commands from the companion Android app via a lightweight character device.

/* 
 * INJECTION TARGET: drivers/char/nexussu.c
 * Implements an ioctl gateway for user-space companion communication
 */

#include <linux/fs.h>
#include <linux/uaccess.h>
#include <linux/nexussu.h>

#define NEXUSSU_IOC_MAGIC 'N'
#define NEXUSSU_ALLOW_UID _IOW(NEXUSSU_IOC_MAGIC, 1, uid_t)
#define NEXUSSU_GET_VERSION _IOR(NEXUSSU_IOC_MAGIC, 2, int)

const int ENGINE_VERSION = 100; // v1.0.0

/* Handle incoming configuration requests from the Manager App */
static long nexussu_ioctl(struct file *file, unsigned int cmd, unsigned long arg)
{
    uid_t target_uid;
    
    switch (cmd) {
        case NEXUSSU_ALLOW_UID:
            if (copy_from_user(&target_uid, (uid_t __user *)arg, sizeof(uid_t))) {
                return -EFAULT;
            }
            /* Explicitly wire up the state storage backend */
            nexussu_add_trusted_uid(target_uid);
            printk(KERN_INFO "NexusSU: Management app added UID %d to allowlist\n", target_uid);
            break;

        case NEXUSSU_GET_VERSION:
            if (copy_to_user((int __user *)arg, &ENGINE_VERSION, sizeof(int))) {
                return -EFAULT;
            }
            break;

        default:
            return -ENOTTY;
    }
    return 0;
}

/* Bind our custom ioctl gateway into standard file operations */
static const struct file_operations nexussu_fops = {
    .owner = THIS_MODULE,
    .unlocked_ioctl = nexussu_ioctl,
    .compat_ioctl = nexussu_ioctl,
};

Step 5: Allowlist Core Engine (kernel/nexussu_state.c)

This component maintains a thread-safe list of authorized Android User IDs (UIDs) using a standard Linux spinlock to protect against race conditions during high-concurrency requests.

+ kernel/nexussu_state.c 

/* 
 * INJECTION TARGET: kernel/nexussu_state.c
 * Implements the backend storage and validation routines for authorized UIDs
 */

#include <linux/spinlock.h>
#include <linux/uidgid.h>
#include <linux/types.h>
#include <linux/slab.h>
#include <linux/string.h>
#include <linux/fs.h>

#define MAX_TRACKED_UIDS 256

static uid_t nexussu_trusted_uids[MAX_TRACKED_UIDS];
static int nexussu_uid_count = 0;
static DEFINE_SPINLOCK(nexussu_lock);

/* Validates if a UID is authorized */
bool nexussu_is_granted(kuid_t kuid)
{
    int i;
    uid_t uid = kuid.val;
    unsigned long flags;
    bool found = false;

    spin_lock_irqsave(&nexussu_lock, flags);
    for (i = 0; i < nexussu_uid_count; i++) {
        if (nexussu_trusted_uids[i] == uid) {
            found = true;
            break;
        }
    }
    spin_unlock_irqrestore(&nexussu_lock, flags);
    return found;
}

/* Adds a new trusted UID to the in-memory array */
void nexussu_add_trusted_uid(uid_t uid)
{
    int i;
    unsigned long flags;

    spin_lock_irqsave(&nexussu_lock, flags);
    for (i = 0; i < nexussu_uid_count; i++) {
        if (nexussu_trusted_uids[i] == uid) {
            spin_unlock_irqrestore(&nexussu_lock, flags);
            return;
        }
    }

    if (nexussu_uid_count < MAX_TRACKED_UIDS) {
        nexussu_trusted_uids[nexussu_uid_count] = uid;
        nexussu_uid_count++;
    }
    spin_unlock_irqrestore(&nexussu_lock, flags);
}

/* Helper logic for 4.x/5.x VFS direct query filters */
bool nexussu_stealth_check(const char __user *filename) {
    char buf[128];
    long len;

    if (nexussu_is_granted(current_uid())) return false;
    if (!filename) return false;

    len = strncpy_from_user(buf, filename, sizeof(buf) - 1);
    if (len <= 0) return false;
    buf[len] = '\0';

    if (strstr(buf, "/su") || strstr(buf, "nexussu") || strstr(buf, "magisk")) {
        return true;
    }
    return false;
}

/* Engine backend for Procfs deep table scrubbing */
void nexussu_scrub_proc_buffer(struct file *file, char __user *buf, size_t count, ssize_t *ret) {
    char *kbuf;
    ssize_t read_bytes = *ret;
    struct dentry *dentry;

    if (read_bytes <= 0 || !buf || !file) return;
    dentry = file->f_path.dentry;
    if (!dentry || !dentry->d_name.name) return;

    if (strcmp(dentry->d_name.name, "mountinfo") == 0 || strcmp(dentry->d_name.name, "mounts") == 0) {
        kbuf = kmalloc(read_bytes + 1, GFP_KERNEL);
        if (!kbuf) return;

        if (copy_from_user(kbuf, buf, read_bytes)) {
            kfree(kbuf);
            return;
        }
        kbuf[read_bytes] = '\0';

        char *line = kbuf, *next_line, *write_ptr = kbuf;
        bool modified = false;

        while (line && *line) {
            next_line = strchr(line, '\n');
            if (next_line) { *next_line = '\0'; next_line++; }

            if (strstr(line, "nexussu") || strstr(line, "magisk") || strstr(line, "/su")) {
                modified = true;
            } else {
                size_t line_len = strlen(line);
                memmove(write_ptr, line, line_len);
                write_ptr += line_len;
                if (next_line) { *write_ptr = '\n'; write_ptr++; }
            }
            line = next_line;
        }

        if (modified) {
            ssize_t new_len = write_ptr - kbuf;
            if (!copy_to_user(buf, kbuf, new_len)) *ret = new_len;
        }
        kfree(kbuf);
    }
}
